### Comments

```c
#include <stdio.h>

int main(){
	// This is a comment
	/*
	This
	is
	a 
	multiline
	comment
	*/
	printf("Hello World!\n")
	printf("This is my first program")
	return 0;
}
```

### Escape Sequences 

```c
#include <stdio.h>

int main(){

	/* Escape sequence = charcter combination consisting of a backslash \  
						followed by a letter or combination of digits.
						They specify actions within a line or string of text.
						\n = newline
						\t = tab


	*/

	printf("1\t2\t3\t\n4\t5\t6\n7\t8\t9")
	return 0;
}
```
